
---CREATE TABLES "STUDENTS AND SUBJECTS"
CREATE TABLE Students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    age INT,
    gender VARCHAR(10),
    course VARCHAR(50)
);
CREATE TABLE Subjects(
subject_id INT PRIMARY KEY,
    subject_name VARCHAR(50),
    student_id INT,
    FOREIGN KEY (student_id) REFERENCES Students(student_id)
);

--- JOINS
select s.name,s.course,sb.subject_name from Students s
 join Subjects sb
 on s.student_id = sb.student_id;

select s.name,s.course,sb.subject_name from Students s
 left join Subjects sb
 on s.student_id = sb.student_id
 order by name asc;

select s.name,s.course, count(sb.subject_name) from Students s
 right join Subjects sb
 on s.student_id = sb.student_id
 group by s.name,s.course
 order by name asc;
